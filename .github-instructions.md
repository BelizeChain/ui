## ğŸ¨ Standard Page Header Pattern (CRITICAL)

**ALL platform pages MUST use this consistent sticky header pattern**:

```tsx
'use client';
import { useRouter } from 'next/navigation';
import { ArrowLeft, PageIcon } from 'phosphor-react';

export default function PageName() {
  const router = useRouter();
  
  return (
    <div className="min-h-screen bg-gradient-to-b from-gray-900 via-gray-800 to-gray-900 pb-24">
      {/* Sticky Header with Back Button */}
      <div className="sticky top-0 bg-gray-900/80 backdrop-blur-xl px-6 py-4 z-10 border-b border-gray-700/50">
        <div className="flex items-center justify-between p-4">
          <div className="flex items-center gap-3">
            <button onClick={() => router.back()} className="p-2 hover:bg-gray-800 rounded-full transition-colors">
              <ArrowLeft size={24} className="text-gray-300" weight="bold" />
            </button>
            <div>
              <h1 className="text-xl font-bold text-white">Page Title</h1>
              <p className="text-xs text-gray-400">Subtitle</p>
            </div>
          </div>
          <div className="flex items-center gap-3">
            {/* Optional badges/icons */}
            <PageIcon size={32} className="text-blue-400" weight="duotone" />
          </div>
        </div>
      </div>

      {/* Main Content Container */}
      <div className="p-4 space-y-6">
        {/* ALL page content goes here */}
        <GlassCard variant="dark-medium" blur="lg" className="p-6">
          {/* Card content */}
        </GlassCard>
      </div>
    </div>
  );
}
```

**Critical Rules**:
- âŒ **NEVER** use full-width gradient GlassCard headers
- âœ… **ALWAYS** use sticky header with back button navigation
- âœ… **ALL** content must be inside `<div className="p-4 space-y-6">`
- âœ… Import `useRouter` from `'next/navigation'` and `ArrowLeft` from `'phosphor-react'`
- âœ… Verify opening/closing `<div>` tags match (common error: missing closing div)

**Pages Using This Pattern** (Reference Implementations):
- âœ… About, Help, Activity (perfect examples)
- âœ… BelizeID, Pakit, Nawal, Kinich, Gem (recently updated)

### API Connection Hook

```typescript
// ui/shared/hooks/usePolkadotAPI.ts
import { ApiPromise, WsProvider } from '@polkadot/api';
import { useEffect, useState } from 'react';

export function usePolkadotAPI(rpcUrl: string = 'ws://localhost:9944') {
  const [api, setApi] = useState<ApiPromise | null>(null);
  const [isReady, setIsReady] = useState(false);
  const [error, setError] = useState<Error | null>(null);

  useEffect(() => {
    let isMounted = true;
    let apiInstance: ApiPromise | null = null;

    async function connectAPI() {
      try {
        const provider = new WsProvider(rpcUrl);
        apiInstance = await ApiPromise.create({ provider });

        if (isMounted) {
          setApi(apiInstance);
          setIsReady(true);
        }
      } catch (err) {
        if (isMounted) {
          setError(err as Error);
        }
      }
    }

    connectAPI();

    return () => {
      isMounted = false;
      if (apiInstance) {
        apiInstance.disconnect();
      }
    };
  }, [rpcUrl]);

  return { api, isReady, error };
}
```

### Extrinsic Submission Hook

```typescript
// ui/shared/hooks/useExtrinsic.ts
import { web3FromSource } from '@polkadot/extension-dapp';
import type { SubmittableExtrinsic } from '@polkadot/api/types';

export function useExtrinsic() {
  const { api } = usePolkadotAPI();

  async function submitExtrinsic(
    extrinsic: SubmittableExtrinsic<'promise'>,
    accountAddress: string,
    injectorSource: string = 'polkadot-js'
  ) {
    if (!api) throw new Error('API not connected');

    // Get injector from Polkadot.js extension
    const injector = await web3FromSource(injectorSource);

    // Sign and send
    return new Promise((resolve, reject) => {
      extrinsic
        .signAndSend(
          accountAddress,
          { signer: injector.signer },
          ({ status, events, dispatchError }) => {
            if (status.isInBlock) {
              console.log(`Included in block: ${status.asInBlock.toHex()}`);
            }

            if (status.isFinalized) {
              console.log(`Finalized in block: ${status.asFinalized.toHex()}`);

              // Check for errors
              if (dispatchError) {
                if (dispatchError.isModule) {
                  const decoded = api.registry.findMetaError(dispatchError.asModule);
                  reject(new Error(`${decoded.section}.${decoded.name}: ${decoded.docs}`));
                } else {
                  reject(new Error(dispatchError.toString()));
                }
              } else {
                resolve({ blockHash: status.asFinalized, events });
              }
            }
          }
        )
        .catch(reject);
    });
  }

  return { submitExtrinsic };
}
```

### Real-Time Subscriptions

```typescript
// ui/shared/hooks/useSubscription.ts
import { useEffect, useState } from 'react';
import { usePolkadotAPI } from './usePolkadotAPI';

export function useBalanceSubscription(accountAddress: string) {
  const { api, isReady } = usePolkadotAPI();
  const [balance, setBalance] = useState<{ free: string; reserved: string } | null>(null);

  useEffect(() => {
    if (!api || !isReady || !accountAddress) return;

    let unsubscribe: (() => void) | undefined;

    // Subscribe to balance changes
    api.query.system.account(accountAddress, (account: any) => {
      setBalance({
        free: account.data.free.toString(),
        reserved: account.data.reserved.toString()
      });
    }).then(unsub => {
      unsubscribe = unsub;
    });

    return () => {
      if (unsubscribe) unsubscribe();
    };
  }, [api, isReady, accountAddress]);

  return balance;
}
```

// ui/shared/components/GlassCard.tsx
import React from 'react';

type GlassCardVariant = 
  | 'light'          // Light background, minimal blur
  | 'dark'           // Dark background, medium blur
  | 'dark-medium'    // Darker background, heavy blur
  | 'gradient';      // Gradient background

type BlurStrength = 'sm' | 'md' | 'lg' | 'xl';

interface GlassCardProps {
  variant?: GlassCardVariant;
  blur?: BlurStrength;
  className?: string;
  children: React.ReactNode;
}

export function GlassCard({
  variant = 'dark',
  blur = 'md',
  className = '',
  children
}: GlassCardProps) {
  const baseClasses = 'rounded-xl border transition-all duration-300';
  
  const variantClasses = {
    light: 'bg-white/10 border-white/20',
    dark: 'bg-gray-800/50 border-gray-700/50',
    'dark-medium': 'bg-gray-900/70 border-gray-700/30',
    gradient: 'bg-gradient-to-br from-blue-500/20 to-purple-500/20 border-blue-400/30'
  };
  
  const blurClasses = {
    sm: 'backdrop-blur-sm',
    md: 'backdrop-blur-md',
    lg: 'backdrop-blur-lg',
    xl: 'backdrop-blur-xl'
  };

  return (
    <div
      className={`${baseClasses} ${variantClasses[variant]} ${blurClasses[blur]} ${className}`}
    >
      {children}
    </div>
  );
}

// ui/shared/utils/formatting.ts
import { formatBalance } from '@polkadot/util';

// Configure DALLA (12 decimals)
formatBalance.setDefaults({
  decimals: 12,
  unit: 'DALLA'
});

export function formatDALLA(amount: string | number): string {
  // Input: "1000000000000" (1 DALLA in base units)
  // Output: "1.00 DALLA"
  return formatBalance(amount, { withSi: false, forceUnit: 'DALLA' });
}

export function formatbBZD(amount: string | number): string {
  // bBZD uses 2 decimals (like USD)
  const value = typeof amount === 'string' ? parseFloat(amount) : amount;
  return `$${(value / 100).toFixed(2)} bBZD`;
}

export function formatUSD(dallaAmount: string | number, exchangeRate: number): string {
  // Convert DALLA â†’ USD for display
  const dallaValue = typeof dallaAmount === 'string' ? parseFloat(dallaAmount) : dallaAmount;
  const usdValue = (dallaValue / 1e12) * exchangeRate;
  return `$${usdValue.toFixed(2)} USD`;
}

import { useExtrinsic } from '@/shared/hooks/useExtrinsic';
import { GlassCard } from '@/shared/components/GlassCard';

  index: number;
  hash: string;
  author: string;
  deposit: string;
  description: string;
  voteAye: number;
  voteNay: number;

export function GovernanceVoting({ proposal }: { proposal: Proposal }) {
  const { submitExtrinsic } = useExtrinsic();
  const [isVoting, setIsVoting] = useState(false);

  async function vote(aye: boolean) {
    setIsVoting(true);
    
    try {
      const extrinsic = api.tx.governance.vote(proposal.index, aye);
      
      await submitExtrinsic(extrinsic, currentAccount.address);
      
      // Show success toast

// ui/maya-wallet/components/StakingDashboard.tsx
import { useEffect, useState } from 'react';
import { usePolkadotAPI } from '@/shared/hooks/usePolkadotAPI';

export function StakingDashboard({ account }: { account: string }) {
  const { api } = usePolkadotAPI();
  const [stakingInfo, setStakingInfo] = useState<any>(null);

  useEffect(() => {
    if (!api || !account) return;

    async function fetchStakingInfo() {
      // Query staking pallet
      const [staker, nominations, rewards] = await Promise.all([
        api.query.staking.stakers(account),
        api.query.staking.nominations(account),
        api.query.staking.pendingRewards(account)
      ]);

      setStakingInfo({
        totalStaked: staker.total.toString(),
        activeStake: staker.active.toString(),
        nominations: nominations.toJSON(),
        pendingRewards: rewards.toString()
      });
    }

    fetchStakingInfo();
  }, [api, account]);

  if (!stakingInfo) return <div>Loading...</div>;

  return (
    <div className="space-y-4">
      <GlassCard variant="gradient" blur="lg" className="p-6">
        <h3 className="text-sm text-gray-400 mb-1">Total Staked</h3>

        <p className="text-2xl font-bold text-green-400">
          {formatDALLA(stakingInfo.pendingRewards)}

// ui/shared/hooks/useAccount.ts
import { web3Accounts, web3Enable } from '@polkadot/extension-dapp';
import { useState, useEffect } from 'react';

export function useAccounts() {
  const [accounts, setAccounts] = useState<any[]>([]);
  const [selectedAccount, setSelectedAccount] = useState<any>(null);

  useEffect(() => {
    async function loadAccounts() {
      // Enable Polkadot.js extension
      await web3Enable('BelizeChain Maya Wallet');
      
      // Get all accounts
      const allAccounts = await web3Accounts();
      setAccounts(allAccounts);
      
      // Select first account by default
      if (allAccounts.length > 0) {
        setSelectedAccount(allAccounts[0]);
      }
    }

    loadAccounts();
  }, []);

  return { accounts, selectedAccount, setSelectedAccount };
}
